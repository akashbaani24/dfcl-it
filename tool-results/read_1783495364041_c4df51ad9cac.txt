   519→   519→// Wrapper that reads config from sessionStorage and renders GenericAddEditPage
   520→   520→function GenericAddEditPageWrapper() {
   521→   521→  const config = typeof window !== 'undefined' ? sessionStorage.getItem('genericAddEditConfig') : null
   522→   522→  if (!config) return <div className="p-6 text-sm text-muted-foreground">No config found. Go back.</div>
   523→   523→  // Parse config outside of any JSX construction so render-time errors are not
   524→   524→  // swallowed by a try/catch (which would violate react-hooks/error-boundaries).
   525→   525→  let parsed: { slug: string; title: string; fields: any[]; defaultValues?: Record<string, any>; backTo: any } | null = null
   526→   526→  try {
   527→   527→    parsed = JSON.parse(config)
   528→   528→  } catch {
   529→   529→    return <div className="p-6 text-sm text-muted-foreground">Invalid config.</div>
   530→   530→  }
   531→   531→  if (!parsed) return <div className="p-6 text-sm text-muted-foreground">Invalid config.</div>
   532→   532→  const { slug, title, fields, defaultValues, backTo } = parsed
   533→   533→  return (
   534→   534→    <GenericAddEditPage
   535→   535→      slug={slug}
   536→   536→      title={title}
   537→   537→      fields={fields}
   538→   538→      defaultValues={defaultValues}
   539→   539→      backTo={backTo}
   540→   540→    />
   541→   541→  )
   542→   542→}
